128  /**
129   * Returns an unmodifiable, non-{@code null} {@link Collection} of
130   * {@link Artifact}s, each element of which is a non-{@code null}
131   * {@link Artifact} that represents either the supplied {@link
132   * MavenProject} itself or a {@linkplain MavenProject#getArtifacts()
133   * dependency of it}.
134   *
135   * <p>The returned {@link Artifact} {@link Collection} will be
136   * sorted in topological order, from an {@link Artifact} with no
137   * dependencies as the first element, to the {@link Artifact}
138   * representing the {@link MavenProject} itself as the last.</p>
139   *
140   * <p>All {@link Artifact}s that are not {@linkplain
141   * Object#equals(Object) equal to} the return value of {@link
142   * MavenProject#getArtifact() project.getArtifact()} will be
143   * {@linkplain ArtifactResolver#resolve(ArtifactResolutionRequest)
144   * resolved} if they are not already {@linkplain
145   * Artifact#isResolved() resolved}.  No guarantee of {@linkplain
146   * Artifact#isResolved() resolution status} is made of the
147   * {@linkplain MavenProject#getArtifact() project
148   * <code>Artifact</code>}, which in normal&mdash;possibly
149   * all?&mdash;cases will be unresolved.</p>
150   *
151   * @param project the {@link MavenProject} for which resolved {@link
152   * Artifact}s should be returned; must not be {@code null}
153   *
154   * @param dependencyGraphBuilder the {@link DependencyGraphBuilder}
155   * instance that will calculate the dependency graph whose postorder
156   * traversal will yield the {@link Artifact}s to be resolved; must
157   * not be {@code null}
158   *
159   * @param filter an {@link ArtifactFilter} used during {@linkplain
160   * DependencyGraphBuilder#buildDependencyGraph(MavenProject,
161   * ArtifactFilter) dependency graph assembly}; may be {@code null}
162   *
163   * @param resolver an {@link ArtifactResolver} that will use the <a
164   * href="http://maven.apache.org/ref/3.0.5/maven-compat/apidocs/src-html/org/apache/maven/artifact/resolver/DefaultArtifactResolver.html#line.335">Maven
165   * 3.0.5 <code>Artifact</code> resolution algorithm</a> to resolve
166   * {@link Artifact}s returned by the {@link
167   * DependencyGraphBuilder#buildDependencyGraph(MavenProject,
168   * ArtifactFilter)} method; must not be {@code null}
169   *
170   * @param localRepository an {@link ArtifactRepository} representing
171   * the local Maven repository in effect; may be {@code null}
172   *
173   * @return a non-{@code null}, {@linkplain
174   * Collections#unmodifiableCollection(Collection) unmodifiable}
175   * {@link Collection} of non-{@code null} {@link Artifact} instances
176   *
177   * @exception IllegalArgumentException if {@code project}, {@code
178   * dependencyGraphBuilder} or {@code resolver} is {@code null}
179   *
180   * @exception ArtifactResolutionException if there were problems
181   * {@linkplain ArtifactResolver#resolve(ArtifactResolutionRequest)
182   * resolving} {@link Artifact} instances
183   *
184   * @exception DependencyGraphBuilderException if there were problems
185   * {@linkplain
186   * DependencyGraphBuilder#buildDependencyGraph(MavenProject,
187   * ArtifactFilter) building the dependency graph}
188   *
189   * @see ArtifactResolver#resolve(ArtifactResolutionRequest)
190   *
191   * @see DependencyGraphBuilder#buildDependencyGraph(MavenProject,
192   * ArtifactFilter)
193   */
194  public Collection<? extends Artifact> getArtifactsInTopologicalOrder(final MavenProject project,
195                                                                       final DependencyGraphBuilder dependencyGraphBuilder, 
196                                                                       final ArtifactFilter filter,
197                                                                       final ArtifactResolver resolver,
198                                                                       final ArtifactRepository localRepository)
199    throws DependencyGraphBuilderException, ArtifactResolutionException {
200    final Logger logger = this.getLogger();
201    if (logger != null && logger.isLoggable(Level.FINER)) {
202      logger.entering(this.getClass().getName(), "getArtifactsInTopologicalOrder", new Object[] { project, dependencyGraphBuilder, filter, resolver, localRepository });
203    }
204    if (project == null) {
205      throw new IllegalArgumentException("project", new NullPointerException("project"));
206    }
207    if (dependencyGraphBuilder == null) {
208      throw new IllegalArgumentException("dependencyGraphBuilder", new NullPointerException("dependencyGraphBuilder"));
209    }
210    if (resolver == null) {
211      throw new IllegalArgumentException("resolver", new NullPointerException("resolver"));
212    }
213
214    List<Artifact> returnValue = null;
215
216    final DependencyNode projectNode = dependencyGraphBuilder.buildDependencyGraph(project, filter);
217    assert projectNode != null;
218    final CollectingDependencyNodeVisitor visitor = new CollectingDependencyNodeVisitor();
219    projectNode.accept(visitor);
220    final Collection<? extends DependencyNode> nodes = visitor.getNodes();
221
222    if (nodes == null || nodes.isEmpty()) {
223      if (logger != null && logger.isLoggable(Level.FINE)) {
224        logger.logp(Level.FINE, this.getClass().getName(), "getArtifactsInTopologicalOrder", "No dependency nodes encountered");
225      }
226
227    } else {
228      final Artifact projectArtifact = project.getArtifact();
229
230      returnValue = new ArrayList<Artifact>();
231
232      for (final DependencyNode node : nodes) {
233        if (node != null) {
234          Artifact artifact = node.getArtifact();
235          if (artifact != null) {
236            
237            if (!artifact.isResolved()) {
238              if (logger != null && logger.isLoggable(Level.FINE)) {
239                logger.logp(Level.FINE, this.getClass().getName(), "getArtifactsInTopologicalOrder", "Artifact {0} is unresolved", artifact);
240              }
241              
242              if (artifact.equals(projectArtifact)) {
243                // First see if the artifact is the project artifact.
244                // If so, then it by definition won't be able to be
245                // resolved, because it's being built.
246                if (logger != null && logger.isLoggable(Level.FINE)) {
247                  logger.logp(Level.FINE, this.getClass().getName(), "getArtifactsInTopologicalOrder", "Artifact {0} resolved to project artifact: {1}", new Object[] { artifact, projectArtifact });
248                }
249                artifact = projectArtifact;
250                
251              } else {
252                // See if the project's associated artifact map
253                // contains a resolved version of this artifact.  The
254                // artifact map contains all transitive dependency
255                // artifacts of the project.  Each artifact in the map
256                // is guaranteed to be resolved.
257                @SuppressWarnings("unchecked")
258                final Map<String, Artifact> artifactMap = project.getArtifactMap();
259                if (artifactMap != null && !artifactMap.isEmpty()) {
260                  final Artifact mapArtifact = artifactMap.get(new StringBuilder(artifact.getGroupId()).append(":").append(artifact.getArtifactId()).toString());
261                  if (mapArtifact != null) {
262                    if (logger != null && logger.isLoggable(Level.FINE)) {
263                      logger.logp(Level.FINE, this.getClass().getName(), "getArtifactsInTopologicalOrder", "Artifact {0} resolved from project artifact map: {1}", new Object[] { artifact, mapArtifact });
264                    }
265                    artifact = mapArtifact;
266                  }
267                }
268                
269                if (!artifact.isResolved()) {
270                  // Finally, perform manual artifact resolution.
271                  final ArtifactResolutionRequest request = new ArtifactResolutionRequest();
272                  request.setArtifact(artifact);
273                  request.setLocalRepository(localRepository);
274                  @SuppressWarnings("unchecked")
275                  final List<ArtifactRepository> remoteRepositories = project.getRemoteArtifactRepositories();
276                  request.setRemoteRepositories(remoteRepositories);
277                  
278                  if (logger != null && logger.isLoggable(Level.FINE)) {
279                    logger.logp(Level.FINE, this.getClass().getName(), "getArtifactsInTopologicalOrder", "Resolving artifact {0} using ArtifactResolutionRequest {1}", new Object[] { artifact, request });
280                  }
281                  
282                  final ArtifactResolutionResult result = resolver.resolve(request);
283                  if (result == null || !result.isSuccess()) {
284                    this.handleArtifactResolutionError(request, result);
285                  } else {
286                    @SuppressWarnings("unchecked")
287                    final Collection<? extends Artifact> resolvedArtifacts = (Set<? extends Artifact>)result.getArtifacts();
288                    if (resolvedArtifacts == null || resolvedArtifacts.isEmpty()) {
289                      if (logger != null && logger.isLoggable(Level.WARNING)) {
290                        logger.logp(Level.WARNING, this.getClass().getName(), "getArtifactsInTopologicalOrder", "Artifact resolution failed silently for artifact {0}", artifact);
291                      }
292                    } else {
293                      final Artifact resolvedArtifact = resolvedArtifacts.iterator().next();
294                      if (resolvedArtifact != null) {
295                        assert resolvedArtifact.isResolved();
296                        artifact = resolvedArtifact;
297                      } else if (logger != null && logger.isLoggable(Level.WARNING)) {
298                        logger.logp(Level.WARNING, this.getClass().getName(), "getArtifactsInTopologicalOrder", "Artifact resolution failed silently for artifact {0}", artifact);
299                      }
300                    }
301                  }
302                }
303                
304              }
305            }
306            
307            if (artifact != null) {
308              returnValue.add(artifact);
309            }
310          }
311        }
312      }
313      if (!returnValue.isEmpty()) {        
314        Collections.reverse(returnValue);
315        Collections.sort(returnValue, scopeComparator);
316      }
317    }
318    if (returnValue == null) {
319      returnValue = Collections.emptyList();
320    } else {
321      returnValue = Collections.unmodifiableList(returnValue);
322    }
323    if (logger != null && logger.isLoggable(Level.FINER)) {
324      logger.exiting(this.getClass().getName(), "getArtifactsInTopologicalOrder", returnValue);
325    }
326    return returnValue;
327  }

329  /**
330   * Handles an error that is represented by the supplied {@link
331   * ArtifactResolutionResult} that was issued in response to the
332   * supplied {@link ArtifactResolutionRequest}.
333   *
334   * <p>This implementation performs the following operations:</p>
335   *
336   * <blockquote><pre>new {@link
337   * DefaultResolutionErrorHandler#DefaultResolutionErrorHandler()
338   * DefaultResolutionErrorHandler()}.{@link
339   * DefaultResolutionErrorHandler#throwErrors(ArtifactResolutionRequest,
340   * ArtifactResolutionResult) throwErrors(request,
341   * result)};</pre></blockquote>
342   *
343   * @param request the {@link ArtifactResolutionRequest} that caused
344   * the error; must not be {@code null}
345   *
346   * @param result the {@link ArtifactResolutionResult} that resulted;
347   * must not be {@code null}
348   *
349   * @exception ArtifactResolutionException if this method is not
350   * overridden to do something else
351   *
352   * @see DefaultResolutionErrorHandler
353   *
354   * @see
355   * DefaultResolutionErrorHandler#throwErrors(ArtifactResolutionRequest,
356   * ArtifactResolutionResult)
357   */
358  protected void handleArtifactResolutionError(final ArtifactResolutionRequest request, final ArtifactResolutionResult result) throws ArtifactResolutionException {
359    new DefaultResolutionErrorHandler().throwErrors(request, result);
360  }

368  /**
369   * A {@link Comparator} of {@link Artifact}s that sorts {@code
370   * test}-scoped {@link Artifact}s to the "right".
371   *
372   * <p>Note: this {@linkplain Comparator comparator} imposes
373   * orderings that are {@linkplain Comparator#compare(Object, Object)
374   * inconsistent with equals}.</p>
375   *
376   * <p>Instances of this class are safe for use by concurrent {@link
377   * Thread}s.</p>
378   *
379   * @author <a href="http://about.me/lairdnelson"
380   * target="_parent">Laird Nelson</a>
381   *
382   * @see Comparator
383   *
384   * @see Artifact
385   *
386   * @see Artifact#getScope()
387   */
388  private static final class ScopeComparator implements Comparator<Artifact> {
389
390    /**
391     * Creates a new {@link ScopeComparator}.
392     */
393    private ScopeComparator() {
394      super();
395    }
396
397    /**
398     * Compares the two supplied {@link Artifact}s by testing
399     * {@linkplain Artifact#getScope() their affiliated scopes}.
400     *
401     * @param a1 the first {@link Artifact} to test; must not be
402     * {@code null}
403     *
404     * @param a2 the second {@link Artifact} to test; must not be
405     * {@code null}
406     *
407     * @return {@code 1} when the first {@link Artifact} has a
408     * {@linkplain Artifact#getScope() scope} of {@code test} and the
409     * second one doesn't; {@code -1} when the second {@link Artifact}
410     * has a {@linkplain Artifact#getScope() scope} of {@code test}
411     * and the first one doesn't; {@code 0} in all other cases
412     *
413     * @exception NullPointerException if either {@code a1} or {@code
414     * a2} is {@code null}
415     */
416    @Override
417    public final int compare(final Artifact a1, final Artifact a2) {
418      if (a1 == null) {
419        throw new NullPointerException("a1");
420      }
421      if (a2 == null) {
422        throw new NullPointerException("a2");
423      }      
424      if ("test".equals(a1.getScope())) {
425        if (!"test".equals(a2.getScope())) {
426          return 1; // test scopes sort to the right/come at the end
427        }
428      } else if ("test".equals(a2.getScope())) {
429        return -1;
430      }
431      return 0;
432    }
433    
434  }
